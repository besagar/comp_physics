\begin{equation} \mathcal{L} = \sum\limits^{\infty} \frac{m \dot{\phi_i}^2}{2} - \sum\limits^{\infty} \frac{k (\phi_i - \phi_{i+1})^2}{2} \end{equation}

:

\begin{equation} m\ddot{\phi_i} = k (\phi_{i-1} - \phi_{i}) - k (\phi_i - \phi_{i+1}) \end{equation}

, , $\phi_i(i) \rightarrow \phi(x)$.

\begin{equation} \ddot{\phi} = \frac{k}{m} (\phi_{i-1} + \phi_{i+1} - 2 \phi_{i}) = \frac{k}{m} \phi '' \delta x ^2 \end{equation}

,

\begin{equation} \lim_{\delta x \to 0} \frac{k} {m} \delta x^2 = const = c^2 \end{equation}

:

\begin{equation} \frac{1}{c^2}\partial^2_t\phi-\partial^2_x\phi=0 \end{equation}

\begin{gather} \phi(0,t) = \phi(L,t) = 0\\ \partial_t \phi(x,0) = 0 \end{gather}

\begin{equation} \frac{1}{c^2}\frac{\phi(t_{n-1},x_k)+\phi(t_{n+1},x_k)-2 \phi(t_{n},x_k)}{\delta t ^2} - \frac{\phi(t_{n},x_{k-1})+\phi(t_{n},x_{k+1})-2 \phi(t_{n},x_k)}{\delta x ^2} = 0 \end{equation}

def string ( f0 , t, L, c = 1, Nx = 1000, Nt = 1000 ):
    
    # f0 -  , ..      t = 0
    
    xes = np.linspace( 0 , L , Nx )
    
    tes = np.linspace( 0 , t , Nt )
    
    dx = xes[1] - xes[0]
    dt = tes[1] - tes[0]
    
    f = np.zeros ( ( Nt , Nx ) )
    
    initial = f0 ( xes )
    
    C = ( c * dt / dx ) ** 2 #         .
    
    f[ 0 , :  ] = initial
    f[ 0 , 0  ] = 0
    f[ 0 , -1 ] = 0
    
    f[ 1 , : ] = f [ 0 , : ] 
    
    for i in range ( 1 , Nt - 1 ):
        
        for j in range ( 1 , Nx - 1 ):
            
            
            f[ i + 1 , j ] = 2 * ( 1 - C ) * f [ i , j ] - f [ i - 1 , j ] + C * ( f [ i , j - 1 ] + f [ i , j + 1 ] )
            
    return f
